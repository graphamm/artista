
///مراحل ایجاد گیت

ایجاد پوشه
init ایجاد مخزن
ایجاد یک فایل
add اضافه کردن به لیست ترک ها با دستور 
commit -m کامیت کردن فایل های ترک شده با  دستور 
--------------------------------------------
//ایجاد کانفیگ اولیه
git config --Global user.name Fatemeh
git config --Global user.email FatemehZahedi8081@gmail.com
git config --list

//ایجاد مخزن
git init

//نمایش وضعیت فعلی
git status

//اافه کردن به لیست ترک ها
git add نام فایل

//کامیت کردن
git commit -m توضیحات

//نادیده گرفتن فایل ها
git ignore نام یاپسوند فایل هایی که نمیخواهیم ترک شوند

//نمایش موقعیت فعلی
pwd

//ایجاد پوشه
mkdir نام پوشه

//نمایش فایل ها
ls

//لیست فایل ها و فایل های پنهان
ls-a

//وارد شدن 
cd

//تفاوت فایلی که وارد مرحله استیج شده باآخرین کامیت را نشان میدهد
git diff Heade

//نادیده گرفتن تغییرات و خارج شدن از استیج
git reset نام فایل

//لیست برنچ های موجود
git branch

//ایجاد برنچ 
git branch نام برنچ

//خارج شدن از برنچ فعال و وارد شدن به برنچ جدید برای اعمال تغییرات روی برنچ جدید
git checkout نام برنچ جدید

//تغییر نام برنچ
git branch -m نام جدید 

//حذف برنچ
git branch -d نام برنچ

// ادغام کردن برنچ ها 
//اگر بخواهیم تغییراتی که در برنچ اول ایجاد کردیم را با برنچ مستر ادغام کنیم ابتدا وارد برنچ مستر میشویم و از دستور مرج برای ادغام استفاده میکنیم
git checkout master 
git merg نام برنچ اول

//وارد شدن به فایل 
vi نام فایل 

//نوشتن درون فایل 
i

//خارج شدن از فایل
esc:wp

//نمایش فایل و اطلاعات آن 
cat نام فایل 

//نمایش تاریخچه کارها
git log

//شناسایی کامیت و دسترسی به فایلی که ایجاد شده 
git show  شماره فایلی که لاگ کردیم 

//تعداد کامیت ها و مشخصات دوتای اخر را نشان میدهد 
git log -n 2

//از تاریخ که گفتیم به بعد کامیت ها را نشان میدهد
git log --after="2024-12-30" --before="2024-3-15"

//کامیت های کاربری که مینویسیم را نشان میدهد
git outher="nasrin"

//کامیت هایی که فایل به نام اف درونش تغییرات ایجاد شده را نمایش میدهد
git log =follow f

//کامیت هایی ک درونش کلمه اد هست را نمایش میدهد
git log --grep="add"

//مشخصات کامیت را به صورت خلاصه در یک خط نمایش میدهد
git log --oneline

//نمایش برچسب و تگ های ایجاد شده
git tag

//تگ کردن به کامیت با یک نام
* پیغام است-m 
git tag -av1 -m "نام"

//نمایش مشخصات کامل تگ
* اسم تگ v1
git show v1 

//وارد شدن به برنچ وی یک و ایجاد یک شاخه به نام برنچ وی یک
git checkout -b v1 branch v1

//نمایش خط به خط تغییرات فایل 
git blame نام فایل

//با جستجوی دودویی خطاها را از بین کامیت ها پیدا میکند
git bisect start 

//تعیین جایی که کامیت درست کار میکرده را نمایش میدهد
git bisect good شماره کامیت

//تعیین جایی که کامیت مشکل داشته
git bisect bad شماره کامیت

* با جستجوی دودویی بین مراحل خب مرحله ای که خطا داشته را نمایش میدهد

//خارج شدن از حالت بایسکت
git bisect reset
